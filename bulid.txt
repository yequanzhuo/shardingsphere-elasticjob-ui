先 create schema ELASTIC_JOB

打包编译：
mvn clean install -Prelease -DskipTests

cd /opt/software/shardingsphere-elastic-ui-3.0.1

sudo cp /Users/yequanzhuo/IdeaProjects/github/shardingsphere-elasticjob-ui/shardingsphere-elasticjob-ui-distribution/shardingsphere-elasticjob-lite-ui-bin-distribution/target/apache-shardingsphere-elasticjob-3.0.1-lite-ui-bin.tar.gz  ./
